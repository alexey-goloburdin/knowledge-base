# Все книги в базе

```dataviewjs
const formatDate = date => {
    return new Date(date.ts).toLocaleDateString('ru-RU', {
        day: 'numeric',
        month: 'long',
        year: 'numeric'
	});
}

let book_pages = dv.pages("").where(page => {
    if (page["Тип"] && page["Тип"].path == "types/Книга.md" && page.file.folder != "templates") return page;
}).sort(p => p["Закончил читать"], "desc");

dv.table(
    ["Название", "Прочитана", "Закончил читать"],
    book_pages.map(p => {
        return [
			dv.el("a", p["Название"], {href: p.file.path}),
            p.Прочитана ? "✓" : "",
            p["Закончил читать"] !== null ?
            formatDate(p["Закончил читать"]) :
            ""
        ];
    })
);
```
